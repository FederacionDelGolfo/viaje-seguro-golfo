    <script>
        let currentTrip = null;
        let participants = [];
        let attendanceHistory = [];
        let password = "federacion2026";
        let isLoggedIn = false;
        let videoStream = null;

        // Cargar datos guardados
        function loadData() {
            const saved = localStorage.getItem('viajeSeguroData');
            if (saved) {
                const data = JSON.parse(saved);
                currentTrip = data.trip;
                participants = data.participants || [];
                attendanceHistory = data.history || [];
            }
        }

        function saveData() {
            if (currentTrip) {
                localStorage.setItem('viajeSeguroData', JSON.stringify({
                    trip: currentTrip,
                    participants: participants,
                    history: attendanceHistory
                }));
            }
        }

        function login() {
            const inputPass = document.getElementById('password').value;
            if (inputPass === password) {
                isLoggedIn = true;
                document.getElementById('login-screen').style.display = 'none';
                document.getElementById('app-screen').style.display = 'block';
                loadData();
                if (currentTrip) document.getElementById('trip-name').textContent = currentTrip.title;
                renderAll();
            } else {
                alert("Contraseña incorrecta");
            }
        }

        function logout() {
            if (confirm("¿Cerrar sesión?")) {
                location.reload();
            }
        }

        function showTab(n) {
            document.querySelectorAll('.tab-content').forEach(tab => tab.style.display = 'none');
            document.getElementById(`tab-${n}`).style.display = 'block';
            document.querySelectorAll('.tab').forEach((tab, i) => tab.classList.toggle('active', i === n));
        }

        function saveTrip() {
            const title = document.getElementById('tripTitle').value || "Viaje Federación del Golfo";
            currentTrip = { title, date: new Date().toISOString() };
            document.getElementById('trip-name').textContent = title;
            saveData();
            alert("Viaje guardado");
        }

        function addParticipant() {
            if (!currentTrip) return alert("Primero guarda el viaje");
            const name = document.getElementById('name').value.trim();
            if (!name) return alert("Ingresa nombre");
            participants.push({
                id: Date.now(),
                name,
                phone: document.getElementById('phone').value || "-",
                present: false,
                lastLocation: null
            });
            document.getElementById('name').value = '';
            document.getElementById('phone').value = '';
            renderParticipants();
            saveData();
        }

        function renderParticipants() {
            const container = document.getElementById('participants-list');
            container.innerHTML = participants.map(p => `
                <div class="participant">
                    <div><strong>\( {p.name}</strong><br><small> \){p.phone}</small></div>
                    <div>
                        <button onclick="showQR(${p.id})">QR</button>
                        <button onclick="togglePresent(\( {p.id})"> \){p.present ? '✅' : '⏳'}</button>
                    </div>
                </div>
            `).join('');
            document.getElementById('count').textContent = participants.length;
        }

        function togglePresent(id) {
            const p = participants.find(x => x.id === id);
            if (p) p.present = !p.present;
            renderParticipants();
            saveData();
        }

        function showQR(id) {
            const p = participants.find(x => x.id === id);
            const modal = document.createElement('div');
            modal.style.cssText = 'position:fixed;top:0;left:0;width:100%;height:100%;background:rgba(0,0,0,0.85);z-index:9999;display:flex;align-items:center;justify-content:center;';
            modal.innerHTML = `
                <div style="background:white;padding:20px;border-radius:12px;text-align:center;max-width:90%;">
                    <h3>Check-in: ${p.name}</h3>
                    <div id="qrcode-modal"></div>
                    <p><strong>${p.phone}</strong></p>
                    <button onclick="this.closest('.modal-container').remove()">Cerrar</button>
                </div>
            `;
            modal.className = 'modal-container';
            document.body.appendChild(modal);
            setTimeout(() => new QRCode(document.getElementById("qrcode-modal"), `VS|\( {p.id}| \){p.name}`), 50);
        }

        function takeAttendance() {
            const now = new Date().toLocaleString('es-MX');
            const presentCount = participants.filter(p => p.present).length;
            
            attendanceHistory.unshift({
                date: now,
                total: participants.length,
                present: presentCount
            });
            
            saveData();
            renderHistory();
            alert(`Pase de lista registrado: \( {presentCount}/ \){participants.length} presentes`);
        }

        function renderHistory() {
            const container = document.getElementById('attendance-history');
            container.innerHTML = `<h3>Historial</h3>` + attendanceHistory.map(h => `
                <div class="participant">
                    <div>${h.date}</div>
                    <div>\( {h.present}/ \){h.total} presentes</div>
                </div>
            `).join('');
        }

        function simulateLocations() {
            alert("Ubicaciones actualizadas (simulado).\n\nPara Google Maps real agrega tu API Key.");
            document.getElementById('map').innerHTML = `<p>✅ Ubicaciones simuladas actualizadas</p>`;
        }

        function exportToExcel() {
            let csv = "Nombre,Teléfono,Estado\n";
            participants.forEach(p => {
                csv += `"\( {p.name}"," \){p.phone}","${p.present ? 'Presente' : 'Ausente'}"\n`;
            });
            const blob = new Blob([csv], { type: 'text/csv' });
            const url = URL.createObjectURL(blob);
            const a = document.createElement('a');
            a.href = url;
            a.download = `Asistentes_${new Date().toISOString().slice(0,10)}.csv`;
            a.click();
        }

        // === ESCÁNER QR ===
        function startScanner() {
            const video = document.getElementById('video');
            navigator.mediaDevices.getUserMedia({ video: { facingMode: "environment" } })
                .then(stream => {
                    videoStream = stream;
                    video.srcObject = stream;
                    scanQR(video);
                })
                .catch(err => alert("Error al acceder a la cámara: " + err.message));
        }

        function stopScanner() {
            if (videoStream) videoStream.getTracks().forEach(track => track.stop());
        }

        function scanQR(video) {
            const canvas = document.createElement('canvas');
            const ctx = canvas.getContext('2d');
            
            const tick = () => {
                if (video.readyState === video.HAVE_ENOUGH_DATA) {
                    canvas.height = video.videoHeight;
                    canvas.width = video.videoWidth;
                    ctx.drawImage(video, 0, 0, canvas.width, canvas.height);
                    const imageData = ctx.getImageData(0, 0, canvas.width, canvas.height);
                    const code = jsQR(imageData.data, imageData.width, imageData.height);
                    
                    if (code) {
                        document.getElementById('scan-result').innerHTML = `✅ Escaneado: ${code.data}<br><strong>Check-in registrado</strong>`;
                        stopScanner();
                        return;
                    }
                }
                requestAnimationFrame(tick);
            };
            tick();
        }

        function renderAll() {
            renderParticipants();
            renderHistory();
        }

        window.onload = () => {
            loadData();
        };
    </script>
